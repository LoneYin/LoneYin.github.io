JS学习笔记———对象的拷贝

发表于 2017-03-22 | 分类于 前端笔记 |

浅拷贝

ES5实现单个目标对象拷贝多个源对象，其实就是封装了Object.assign方法

显然，Object.assign 方法会跳过值为null或者undefined的源对象

const obj = { a: 1, b: 2, c: { d: 3 } };
const shallowCopiedObj = { …obj };
console.log(shallowCopiedObj);

var dad = {
count:[1,2,3],
game:{Inside:true}
};

var kid = shallowCopy(dad);

kid.count.push(4);
kid.count.toString(); // “1,2,3,4”
dad.count.toString(); // “1,2,3,4”
dad.game === kid.game; // true

JSON.parse(JSON.stringify(obj));

function deepCopy (target, source) {
for (var key in source) {
if (Object.prototype.hasOwnProperty.call(source, key)) {
if (typeof(source[key]) === ‘object’) {
target[key] = Array.isArray(source[key]) ? [] : {};
deepCopy(target[key], source[key])
} else {
target[key] = source[key];
}
}
}
}

var mutiArgDeepCopy = function (target) {
target = target || {};
for (var i = 1; i < arguments.length; i++) {
var source = arguments[i];
deepCopy(target, source);
}
return target;
}

var dad = {
count:[1,2,3],
game:{Inside:true}
};

var kid = mutiArgDeepCopy({}, dad);

kid.count.push(4);
kid.count.toString(); // “1,2,3,4”
dad.count.toString(); // “1,2,3”
dad.game === kid.game; // false

```

可以看出我们将子层级的对象也完成了深拷贝， 递归实现深拷贝的话需要注意对象属性循环引用的问题，不加判断的话递归会无穷无尽
